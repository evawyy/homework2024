\begin{problem}\label{pro:4}
  If \(n \mid p-1, n > 1,(a,p)=1\), prove :
  \begin{enumerate}
    \item \(x^n \equiv a (\mod p)\) has solution \(\iff\) \(a^{\frac{p-1}{n}} \equiv 1 (\mod p)\).

\end{problem}
