\begin{problem}\label{pro:1}
  Assume \(p\) is an odd prime, \(a > 1\) is integal.
  Prove:
  \begin{enumerate}
    \item The odd prime factor of \(a^p-1\) is \(a-1\) or \(2px + 1\),
      where \(x\) is integal.
    \item The odd prime factor of \(a^p +1\) is \(a + 1\) or \(2px + 1\),
      where \(x\) is integal.
  \end{enumerate}
\end{problem}
